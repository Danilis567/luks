# Maulkuerfgesetz Referendum in 1937
country_event = {
	id = luxembourg.4
	title = luxembourg.4.t
	desc = luxembourg.4.d
	picture = GFX_report_event_political_gathering
	
	trigger = {
		tag = LUX
		date > 1937.1.1
		not = { has_country_flag = LUX_maulkuerfgesetz_fired }
		not = { ruling_party = communism }
	}
	
	mean_time_to_happen = {
		days = 10
	}
	
	immediate = {
		set_country_flag = LUX_maulkuerfgesetz_fired
	}
	
	option = {
		name = luxembourg.4.a
		trigger = {
			has_country_flag = LUX_resisted_1937_ban
		}
		add_stability = 0.05
		add_popularity = { ideology = communism popularity = 0.05 }
	}
	
	option = {
		name = luxembourg.4.b
		trigger = {
			not = { has_country_flag = LUX_resisted_1937_ban }
		}
		add_stability = -0.10
		set_popularities = {
			communism = 0
			democratic = 70
			neutrality = 28
			fascism = 2
		}
	}
}
